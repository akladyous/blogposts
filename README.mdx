```ts
export default async function BlogPosts() {
  console.group('Blog');
  const posts = await getPostsMeta('blogposts');
  if (!posts) {
    return <div>not posts</div>;
  }

  return (
    <main>
      <h4>Blog Post</h4>
      {posts.map((post, i) => {
        // debugger;
        return (
          <ul className='m-2' key={i}>
            {
              <li className='my-2'>
                <Link href={`blog/${post.id}`}>{post.title}</Link>
              </li>
            }
          </ul>
        );
      })}
    </main>
  );
}
```
